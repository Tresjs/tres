## Usage

```html
<script lang="ts" setup>
  import { useControls, TresLeches } from '@tresjs/leches'

  useControls({
    awiwi: true,
    slider: {
      value: 0.5,
      min: 0,
      max: 1,
      step: 0.01,
    },
    color: '#008080',
  })
</script>

<template>
  <TresLeches />
</template>
```
